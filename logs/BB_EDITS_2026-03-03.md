## SUPABASE BACKEND PLAN (from C1 transfer)

### Schema (6 tables + existing Blob)

```sql
-- profiles: user management
CREATE TABLE profiles (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email TEXT UNIQUE NOT NULL,
  display_name TEXT,
  avatar TEXT DEFAULT 'planet',
  role TEXT DEFAULT 'member',
  tier TEXT DEFAULT 'free',
  created_at TIMESTAMPTZ DEFAULT now()
);

-- blog_posts: replaces Blob blog storage
CREATE TABLE blog_posts (
  id TEXT PRIMARY KEY,
  slug TEXT UNIQUE NOT NULL,
  title TEXT NOT NULL,
  content TEXT NOT NULL,
  excerpt TEXT,
  tags TEXT[] DEFAULT '{}',
  published BOOLEAN DEFAULT false,
  community BOOLEAN DEFAULT false,
  author TEXT,
  avatar TEXT DEFAULT 'planet',
  social TEXT,
  date DATE DEFAULT CURRENT_DATE,
  updated_at TIMESTAMPTZ DEFAULT now()
);

-- community_messages: replaces Blob messages
CREATE TABLE community_messages (
  id TEXT PRIMARY KEY,
  text TEXT NOT NULL,
  value TEXT,
  type TEXT DEFAULT 'reflection',
  created_at TIMESTAMPTZ DEFAULT now()
);

-- dashboard_config: replaces Blob config
CREATE TABLE dashboard_config (
  key TEXT PRIMARY KEY,
  value JSONB NOT NULL,
  updated_at TIMESTAMPTZ DEFAULT now()
);

-- email_subscribers: for Brevo sync
CREATE TABLE email_subscribers (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email TEXT UNIQUE NOT NULL,
  name TEXT,
  source TEXT DEFAULT 'bonus',
  subscribed_at TIMESTAMPTZ DEFAULT now()
);

-- analytics_events: page views, gate conversions
CREATE TABLE analytics_events (
  id BIGSERIAL PRIMARY KEY,
  event TEXT NOT NULL,
  page TEXT,
  metadata JSONB DEFAULT '{}',
  created_at TIMESTAMPTZ DEFAULT now()
);
```
